public class PackagingAndSustainability {

    // Function to calculate the carbon footprint based on distance and weight
    public static double calculateCarbonFootprint(double weight, double distance) {
        double carbonEmissionPerKgKm = 0.02;  // Example factor (kg CO2 per kg per km)
        return weight * distance * carbonEmissionPerKgKm;
    }

    // Function to recommend packaging material (simplified AI model)
    public static String recommendPackaging(double productWeight) {
        if (productWeight < 1) {
            return "Use lightweight recyclable packaging.";
        } else if (productWeight < 5) {
            return "Use biodegradable packaging.";
        } else {
            return "Use reusable packaging.";
        }
    }

    public static void main(String[] args) {
        double productWeight = 2.5;  // in kilograms
        double deliveryDistance = 100;  // in kilometers

        // Calculate carbon footprint
        double carbonFootprint = calculateCarbonFootprint(productWeight, deliveryDistance);
        System.out.println("Carbon footprint for the delivery: " + carbonFootprint + " kg CO2.");

        // Recommend packaging
        String packagingRecommendation = recommendPackaging(productWeight);
        System.out.println("Packaging recommendation: " + packagingRecommendation);
    }
}
