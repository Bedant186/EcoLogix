import java.util.HashMap;

public class WarehouseSharing {

    // Map to store available warehouse spaces and their prices
    private static HashMap<String, Integer> warehouseSpaces = new HashMap<>();

    // Function to add a warehouse space to the marketplace
    public static void addWarehouseSpace(String warehouseID, int pricePerDay) {
        warehouseSpaces.put(warehouseID, pricePerDay);
        System.out.println("Warehouse " + warehouseID + " added with a price of $" + pricePerDay + " per day.");
    }

    // Function to get warehouse space details
    public static String getWarehouseDetails(String warehouseID) {
        if (warehouseSpaces.containsKey(warehouseID)) {
            return "Warehouse " + warehouseID + " is available at $" + warehouseSpaces.get(warehouseID) + " per day.";
        } else {
            return "Warehouse " + warehouseID + " is not available.";
        }
    }

    public static void main(String[] args) {
        // Adding warehouse spaces
        addWarehouseSpace("WH001", 150);
        addWarehouseSpace("WH002", 100);

        // Fetching warehouse details
        System.out.println(getWarehouseDetails("WH001"));  // Output the warehouse details
        System.out.println(getWarehouseDetails("WH003"));  // Output: Warehouse not available
    }
}
