public class InventoryForecasting {

    // Function to simulate demand prediction
    public static double predictDemand(int currentStock, int historicalDemand, int externalFactor) {
        // Basic forecasting formula (for demonstration purposes)
        double predictedDemand = (currentStock + historicalDemand + externalFactor) / 3.0;
        return predictedDemand;
    }

    // Function to check for low stock or overstock
    public static String checkStockLevels(int currentStock, double predictedDemand) {
        if (currentStock < predictedDemand) {
            return "Low Stock: Restock is required.";
        } else if (currentStock > predictedDemand) {
            return "Overstock: Consider reducing inventory.";
        } else {
            return "Stock is optimal.";
        }
    }

    public static void main(String[] args) {
        int currentStock = 50;
        int historicalDemand = 40;
        int externalFactor = 10; // E.g., seasonal trends

        // Predict demand based on current stock and external factors
        double predictedDemand = predictDemand(currentStock, historicalDemand, externalFactor);

        // Check stock levels
        String stockStatus = checkStockLevels(currentStock, predictedDemand);
        System.out.println(stockStatus);  // Output the stock status
    }
}
